```python

	for i in range(how_often):
		print(message)
		
try:
	message = "Everything is awesome!"
	try:
		message = message / 7
	except Exception:
		message = "Everything is horrible!"
		
	number = int("seven")
	repeat_message(message, number)
except Exception:
	print("something went wrong!")
	repeat_message(message, 2)
```
